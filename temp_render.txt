          {activePage === 'Dashboard' && (
          <DashboardPage
            kpis={kpis}
            resultado={{
              receita: monthFinance.receita,
              custos: monthFinance.custos,
              saldo: monthFinance.receita - monthFinance.custos,
              pendReceber: monthFinance.pendReceber,
              pendPagar: monthFinance.pendPagar,
            }}
            proximosFinanceiro={proximosFinanceiro}
            onVerFinanceiro={() => {
              setActivePage('Financeiro')
              setFinanceTab('recebimentos')
              setFinanceSearchPreset('')
            }}
          onKpiClick={(id) => {
            setActivePage('Financeiro')
            if (id === 'custos') {
              setFinanceTab('pagamentos')
            } else {
              setFinanceTab('recebimentos')
            }
            setFinanceSearchPreset('')
          }}
          onShortcut={(label) => {
            const lower = label.toLowerCase()
            if (lower.includes('venda')) setActivePage('Vendas')
            else if (lower.includes('compra')) setActivePage('Compras')
            else if (lower.includes('cliente')) setActivePage('Clientes')
            else if (lower.includes('produto')) setActivePage('Produtos')
            else if (lower.includes('boleto') || lower.includes('conta')) {
              setActivePage('Financeiro')
              setFinanceTab('recebimentos')
            } else {
              setActivePage('Dashboard')
            }
          }}
          dailyNet={dailyNet}
          formatMoney={formatMoney}
        />
      )}
          {activePage === 'Financeiro' && (
            <FinanceiroPage
              activeTab={financeTab}
              setActiveTab={setFinanceTab}
              entries={entries}
              addEntry={addEntry}
              removeEntry={removeEntry}
              updateEntry={updateEntry}
              formatMoney={formatMoney}
              presetSearch={financeSearchPreset}
            />
          )}
          {activePage === 'Vendas' && (
            <VendasPage
              activeTab={salesTab}
              setActiveTab={setSalesTab}
              onFinanceUpsertSale={upsertFinanceFromSale}
              onFinanceRemoveSales={(ids) => removeFinanceByRef(ids.map((id) => financeRefSale(id)))}
              onFinanceMarkSales={(ids) => markFinanceByRef(ids.map((id) => financeRefSale(id)))}
              financeStatusBySale={financeBySale}
              onViewFinance={(id) => {
                setActivePage('Financeiro')
                setFinanceTab(resolveRegistro(salesItems.find((s) => s.id === id) || { registro: 'vendas' }))
                setFinanceSearchPreset(financeRefSale(id))
              }}
            />
          )}
          {activePage === 'Compras' && (
            <ComprasPage
              activeTab={purchaseTab}
              setActiveTab={setPurchaseTab}
              onFinanceUpsertPurchase={upsertFinanceFromPurchase}
              onFinanceRemovePurchases={(ids) => removeFinanceByRef(ids.map((id) => financeRefPurchase(id)))}
              onFinanceMarkPurchases={(ids) => markFinanceByRef(ids.map((id) => financeRefPurchase(id)))}
              financeStatusByPurchase={financeByPurchase}
              onViewFinance={(id) => {
                setActivePage('Financeiro')
                setFinanceTab('pagamentos')
                setFinanceSearchPreset(financeRefPurchase(id))
              }}
            />
          )}
          {activePage === 'Clientes' && <ClientesPage activeTab={contactTab} setActiveTab={setContactTab} />}
          {activePage === 'Produtos' && <ProdutosPage activeTab={productTab} setActiveTab={setProductTab} />}
          {activePage === 'Relatorios' && <RelatoriosPage />}
          {activePage === 'Configuracoes' && <ConfiguracoesPage activeTab={configTab} setActiveTab={setConfigTab} />}
        </main>
      </div>
    </div>
  )
}
